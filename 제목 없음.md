```java
Scanner sc = new Scanner(System.in);

char[] arr = sc.next().toCharArray();
int result = 0;

for (char c : arr) {
  result += (c - 'A') / 3 + 3;
  if (c == 'S' || c == 'V' || c == 'Y' || c == 'Z') {
    result--;
  }
}

sc.close();
System.out.println(result);
```
